魹ｽ魹ｽ魹ｽ魹ｽ $t_n$ 魹ｽﾇｵ魹ｽ $n$ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾇｰ魹ｽﾄｵﾈｴ魹ｽﾊｱ魹ｽ茱ｬ$t_{¥text{base}}$ 魹ｽﾇｻ魹ｽ魹ｽ魹ｽ魹ｽﾓｳﾙ｣魹ｽjitter 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾖｹ魹ｽ魹ｽ魹ｽ魹ｽﾍｻ魹ｽ魹ｽ魹ｽﾍｬﾊｱ魹ｽ魹ｽ魹ｽﾔｵ魹ｽ魹ｽ魹ｽ"魹ｽ魹ｽﾈｺﾐｧﾓｦ"魹ｽ魹ｽ魹ｽ魹ｽ

```python
import asyncio
import random
from functools import wraps
from typing import Callable, Type, Tuple

def retry_with_backoff(
    max_retries: int = 3,
    base_delay: float = 1.0,
    max_delay: float = 60.0,
    retryable_exceptions: Tuple[Type[Exception], ...] = (TimeoutError, ConnectionError),
):
    """ﾖｸ魹ｽ魹ｽ魹ｽﾋｱ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾗｰ魹ｽ魹ｽ魹ｽ魹ｽ"""
    def decorator(func: Callable):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(max_retries + 1):
                try:
                    return await func(*args, **kwargs)
                except retryable_exceptions as e:
                    last_exception = e
                    if attempt == max_retries:
                        break
                    delay = min(base_delay * (2 ** attempt) + random.uniform(0, 1), max_delay)
                    print(f"  [魹ｽ魹ｽ魹ｽ魹ｽ] 魹ｽ魹ｽ {attempt+1} 魹ｽ魹ｽﾊｧ魹ｽ魹ｽ: {e}魹ｽ魹ｽ{delay:.1f}s 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ...")
                    await asyncio.sleep(delay)
            raise last_exception
        return wrapper
    return decorator


# ﾊｹ魹ｽ魹ｽﾊｾ魹ｽ魹ｽ
@retry_with_backoff(max_retries=3, base_delay=1.0)
async def call_search_api(query: str) -> dict:
    """魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ API魹ｽ魹ｽﾊｧ魹ｽ魹ｽ魹ｽﾔｶ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ"""
    async with aiohttp.ClientSession() as session:
        async with session.post(
            "https://api.search.com/v1/search",
            json={"query": query},
            timeout=aiohttp.ClientTimeout(total=10),
        ) as resp:
            if resp.status == 429:
                raise ConnectionError("API 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾒｪ魹ｽﾋｱ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ")
            resp.raise_for_status()
            return await resp.json()
```

```python
import time
from collections import deque
from dataclasses import dataclass, field

@dataclass
class CircuitBreaker:
    """魹ｽ徢･ｵ魹ｽ魹ｽﾛｶ魹ｽ魹ｽ魹ｽﾊｵ魹ｽ魹ｽ"""
    failure_threshold: int = 5          # 魹ｽ魹ｽ魹ｽ魹ｽﾊｧ魹ｽﾜｴ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾖｵ
    recovery_timeout: float = 30.0      # 魹ｽﾛｶﾏｺ魹ｽﾖｸ魹ｽ魹ｽﾈｴ魹ｽﾊｱ魹ｽ茱ｨ魹ｽ槵ｩ
    half_open_max_calls: int = 3        # 魹ｽ歖ｪﾗｴﾌｬ魹ｽ魹ｽ魹ｽﾌｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ

    _state: str = "closed"              # closed / open / half_open
    _failure_count: int = 0
    _last_failure_time: float = 0.0
    _half_open_calls: int = 0

    @property
    def state(self) -> str:
        if self._state == "open":
            if time.time() - self._last_failure_time > self.recovery_timeout:
                self._state = "half_open"
                self._half_open_calls = 0
        return self._state

    def record_success(self):
        if self._state == "half_open":
            self._state = "closed"
        self._failure_count = 0

    def record_failure(self):
        self._failure_count += 1
        self._last_failure_time = time.time()
        if self._state == "half_open":
            self._state = "open"
        elif self._failure_count >= self.failure_threshold:
            self._state = "open"

    def can_execute(self) -> bool:
        state = self.state
        if state == "closed":
            return True
        if state == "half_open":
            if self._half_open_calls < self.half_open_max_calls:
                self._half_open_calls += 1
                return True
            return False
        return False  # open ﾗｴﾌｬ魹ｽﾜｾ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ
```

```python
# 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾊｵ魹ｽ魹ｽ
class ToolExecutorWithFallback:
    """魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾔｵﾄｹ魹ｽ魹ｽ魹ｽﾖｴ魹ｽ魹ｽ魹ｽ魹ｽ"""

    def __init__(self):
        self.cache = {}          # 魹ｽ徢･ｻ魹ｽ魹ｽ魹ｽ
        self.breakers = {}       # ﾃｿ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾒｻ魹ｽ魹ｽ魹ｽﾛｶ魹ｽ魹ｽ魹ｽ

    async def execute_with_fallback(
        self,
        tool_name: str,
        arguments: dict,
        primary_func: Callable,
        fallback_func: Callable = None,
        default_response: str = None,
    ) -> dict:
        """ﾖｴ魹ｽﾐｹ魹ｽ魹ｽﾟ｣魹ｽﾊｧ魹ｽ魹ｽﾊｱ魹ｽ魹ｽ羶ｵ魹ｽ魹ｽ"""
        breaker = self.breakers.setdefault(tool_name, CircuitBreaker())

        # 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ羮ｶ 0魹ｽ魹ｽ魹ｽﾛｶ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾊｱﾖｱ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ
        if not breaker.can_execute():
            return await self._fallback(arguments, fallback_func, default_response)

        try:
            result = await primary_func(**arguments)
            breaker.record_success()
            cache_key = f"{tool_name}:{hash(str(arguments))}"
            self.cache[cache_key] = result
            return {"status": "success", "data": result}

        except Exception as e:
            breaker.record_failure()
            return await self._fallback(arguments, fallback_func, default_response, str(e))

    async def _fallback(self, arguments, fallback_func, default_response, error=None):
        """魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ"""
        # 魹ｽ羮ｶ 2魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾃｹ魹ｽ魹ｽ魹ｽ
        if fallback_func:
            try:
                result = await fallback_func(**arguments)
                return {"status": "degraded", "data": result, "source": "fallback"}
            except Exception:
                pass

        # 魹ｽ羮ｶ 3魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ
        cache_key = f"{hash(str(arguments))}"
        if cache_key in self.cache:
            return {"status": "degraded", "data": self.cache[cache_key], "source": "cache"}

        # 魹ｽ羮ｶ 4魹ｽ魹ｽﾄｬ魹ｽﾏｻﾘｸ魹ｽ
        return {"status": "failed", "data": default_response, "error": error}
```

# ============ 1. 魹ｽﾛｶ魹ｽ魹ｽ魹ｽ ============
@dataclass
class CircuitBreaker:
    failure_threshold: int = 5
    recovery_timeout: float = 30.0
    _state: str = "closed"
    _failures: int = 0
    _last_fail_time: float = 0.0

    def _check_state(self):
        if self._state == "open" and time.time() - self._last_fail_time > self.recovery_timeout:
            self._state, self._failures = "half_open", 0
        return self._state

    def allow(self) -> bool:
        return self._check_state() in ("closed", "half_open")

    def on_success(self):
        self._state, self._failures = "closed", 0

    def on_failure(self):

        self._last_fail_time = time.time()
        if self._failures >= self.failure_threshold:

        timeout: float = 15.0,

        breaker = self.breakers.setdefault(name, CircuitBreaker())

        # --- 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾔｵ魹ｽﾖｴ魹ｽ魹ｽ ---

                result = await asyncio.wait_for(handler(**args), timeout=timeout)
                breaker.on_success()
                self.cache[f"{name}:{args}"] = result
                return {"ok": True, "data": result}

                if attempt < max_retries:

import time
import threading

class TokenBucketRateLimiter:
    """魹ｽ魹ｽ魹ｽ魹ｽﾍｰ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾟｳﾌｰ魹ｽﾈｫ魹ｽ魹ｽﾖｧ魹ｽ魹ｽﾍｻ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ"""

    def __init__(self, rate: float, capacity: int):
        """
        Args:
            rate: 魹ｽ魹ｽ魹ｽﾆｲ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾊ｣魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ/魹ｽ槵ｩ
            capacity: ﾍｰ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾍｻ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ
        """
        self.rate = rate
        self.capacity = capacity
        self.tokens = capacity
        self.last_refill = time.time()
        self._lock = threading.Lock()

    def acquire(self, tokens: int = 1, timeout: float | None = None) -> bool:
        """
        魹ｽ魹ｽ魹ｽﾔｻ魹ｽﾈ｡魹ｽ魹ｽ魹ｽﾆ｡魹ｽ
        魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ timeout魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾈｴ魹ｽﾖｱ魹ｽ魹ｽ魹ｽ魹ｽﾈ｡魹ｽ魹ｽﾊｱ魹ｽ魹ｽ
        """
        deadline = time.time() + timeout if timeout else None
        while True:
            with self._lock:
                self._refill()
                if self.tokens >= tokens:
                    self.tokens -= tokens
                    return True
                # 魹ｽ魹ｽ魹ｽ羹ｹ魹ｽ魹ｽﾈｴ魹ｽ魹ｽ魹ｽﾃｲ魹ｽ魹ｽ魹ｽ魹ｽ羯ｻ魹ｽ魹ｽ魹ｽ魹ｽ
                deficit = tokens - self.tokens
                wait_time = deficit / self.rate
            if deadline and time.time() + wait_time > deadline:
                return False
            if deadline:
                remaining = deadline - time.time()
                time.sleep(min(wait_time, remaining))
            else:
                time.sleep(wait_time)

    def _refill(self):
        now = time.time()
        elapsed = now - self.last_refill
        self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
        self.last_refill = now

```python
import time
import random
import logging
from functools import wraps
from typing import Callable

logger = logging.getLogger("llm.retry")

def retry_with_backoff(
    max_retries: int = 5,
    base_delay: float = 1.0,
    max_delay: float = 60.0,
    retry_on: tuple = (429, 500, 502, 503),
):
    """ﾖｸ魹ｽ魹ｽ魹ｽﾋｱ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾗｰ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ Retry-After ﾍｷ"""
    def decorator(func: Callable) -> Callable:
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_retries + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    status = _get_status_code(e)
                    if status not in retry_on and attempt > 0:
                        raise  # 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾔｵﾄｴ魹ｽ魹ｽ魹ｽﾖｱ魹ｽ魹ｽ魹ｽﾗｳ魹ｽ

                    if attempt == max_retries:
                        logger.error(f"{func.__name__} 魹ｽ魹ｽ魹ｽ魹ｽ {max_retries} 魹ｽﾎｺ魹ｽ魹ｽ魹ｽﾊｧ魹ｽ魹ｽ")
                        raise

                    # 429 魹ｽ魹ｽ魹ｽ魹ｽﾊｹ魹ｽ魹ｽ Retry-After
                    if status == 429:
                        delay = _extract_retry_after(e) or _calc_backoff(
                            attempt, base_delay, max_delay
                        )
                    else:
                        delay = _calc_backoff(attempt, base_delay, max_delay)

                    logger.warning(
                        f"{func.__name__} 魹ｽ魹ｽ {attempt+1} 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾔ｣魹ｽ魹ｽﾈｴ魹ｽ {delay:.1f}s"
                    )
                    time.sleep(delay)
        return wrapper
    return decorator


def _calc_backoff(attempt: int, base: float, max_delay: float) -> float:
    """魹ｽ魹ｽ魹ｽ魹ｽﾖｸ魹ｽ魹ｽ魹ｽﾋｱ魹ｽ + 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ"""
    delay = min(base * (2 ** attempt), max_delay)
    jitter = random.uniform(0, delay * 0.1)  # 10% 魹ｽ魹ｽ魹ｽ魹ｽ
    return delay + jitter


def _get_status_code(error: Exception) -> int | None:
    """魹ｽ魹ｽ魹ｽ瑭｣魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾈ｡ HTTP ﾗｴﾌｬ魹ｽ魹ｽ"""
    status = getattr(error, "status_code", None)
    if status is None and hasattr(error, "response"):
        status = getattr(error.response, "status_code", None)
    return status


def _extract_retry_after(error: Exception) -> float | None:
    """魹ｽﾓｴ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾈ｡ Retry-After ﾖｵ"""
    retry_after = getattr(error, "retry_after", None)
    if retry_after:
        return float(retry_after)
    response = getattr(error, "response", None)
    if response:
        headers = getattr(response, "headers", {})
        ra = headers.get("Retry-After") or headers.get("retry-after")
        if ra:
            try:
                return float(ra)
            except ValueError:
                pass
    return None
```

```python
@dataclass
class AdaptiveState:
    current_rate: float       # 魹ｽ魹ｽﾇｰ魹ｽ魹ｽ魹ｽﾆｲ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ
    min_rate: float           # 魹ｽ魹ｽﾐ｡魹ｽ魹ｽ魹ｽ魹ｽ
    max_rate: float           # 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ
    success_streak: int = 0   # 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾉｹ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ
    rate_limit_hits: int = 0  # 429 魹ｽﾛｼﾆｴ魹ｽ魹ｽ魹ｽ

class AdaptiveRateLimiter:
    """
    魹ｽ魹ｽ魹ｽ魹ｽﾓｦ魹ｽ魹ｽ魹ｽ魹ｽﾍｰ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ 429 ﾆｵ魹ｽﾊｶ魹ｽﾌｬ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾊ｡魹ｽ
    - 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾉｹ魹ｽ魹ｽ羯ｻ魹ｽ魹ｽ魹ｽ 魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾊ｣魹ｽﾌｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾞ｣魹ｽ
    - 魹ｽ魹ｽ魹ｽ魹ｽ 429 魹ｽ魹ｽ 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾊ｣魹ｽ魹ｽ魹ｽ魹ｽﾘｻ魹ｽ魹ｽﾋ｣魹ｽ
    """

    def __init__(
        self,
        initial_rate: float = 5.0,
        min_rate: float = 1.0,
        max_rate: float = 20.0,
        increase_factor: float = 1.2,
        decrease_factor: float = 0.5,
        success_threshold: int = 20,
    ):
        self.state = AdaptiveState(
            current_rate=initial_rate, min_rate=min_rate, max_rate=max_rate,
        )
        self.increase_factor = increase_factor
        self.decrease_factor = decrease_factor
        self.success_threshold = success_threshold
        self.bucket = TokenBucketRateLimiter(
            rate=initial_rate, capacity=int(initial_rate * 2),
        )

    def acquire(self, timeout: float | None = None) -> bool:
        return self.bucket.acquire(tokens=1, timeout=timeout)

    def on_success(self):
        """魹ｽ魹ｽ魹ｽ魹ｽﾉｹ魹ｽﾊｱ魹ｽ魹ｽ魹ｽﾃ｣魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾉｹ魹ｽ魹ｽ髃ｽ魹ｽ魹ｽﾖｵ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ"""
        self.state.success_streak += 1
        if self.state.success_streak >= self.success_threshold:
            new_rate = min(
                self.state.current_rate * self.increase_factor,
                self.state.max_rate,
            )
            if new_rate > self.state.current_rate:
                self.state.current_rate = new_rate
                self.bucket.rate = new_rate
                self.bucket.capacity = int(new_rate * 2)
                logger.info(f"魹ｽ魹ｽ魹ｽ魹ｽﾓｦ魹ｽ魹ｽ魹ｽ魹ｽ: {new_rate:.1f} tokens/s")
            self.state.success_streak = 0

    def on_rate_limit(self):
        """魹ｽﾕｵ魹ｽ 429 ﾊｱ魹ｽ魹ｽ魹ｽﾃ｣魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ"""
        self.state.rate_limit_hits += 1
        self.state.success_streak = 0
        new_rate = max(
            self.state.current_rate * self.decrease_factor,
            self.state.min_rate,
        )
        self.state.current_rate = new_rate
        self.bucket.rate = new_rate
        self.bucket.capacity = int(new_rate * 2)
        logger.warning(
            f"魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ: {new_rate:.1f} tokens/s "
            f"(魹ｽﾛｼ魹ｽ {self.state.rate_limit_hits} 魹ｽ魹ｽ)"
        )
```

        self.semaphore = asyncio.Semaphore(self.config.max_concurrency)
        self.limiter = AdaptiveRateLimiter(
            initial_rate=self.config.initial_rate,
            min_rate=self.config.min_rate,
            max_rate=self.config.max_rate,
        )

    async def chat(
        self,
        messages: list[dict],
        model: str | None = None,
        max_tokens: int = 1024,
    ) -> str:
        """魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾔｵ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ"""

        async with self.semaphore:
            return await self._call_with_retry(messages, model, max_tokens)

    async def _call_with_retry(
        self,
        messages: list[dict],
        model: str,
        max_tokens: int,
    ) -> str:
        for attempt in range(self.config.max_retries):
            self.limiter.acquire(timeout=self.config.request_timeout)
            try:
                # resp = await self.client.chat.completions.create(...)
                self.limiter.on_success()
                return f"response for {model}"

            except Exception as e:
                status = _get_status_code(e)
                if status == 429:
                    self.limiter.on_rate_limit()
                    wait = _extract_retry_after(e) or _calc_backoff(
                        attempt,
                        self.config.base_retry_delay,
                        self.config.max_retry_delay,
                    )
                    logger.warning(f"429 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ {attempt+1} 魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾔ｣魹ｽ魹ｽﾈｴ魹ｽ {wait:.1f}s")
                    await asyncio.sleep(wait)
                elif status in (500, 502, 503):
                    wait = _calc_backoff(attempt, self.config.base_retry_delay,
                                         self.config.max_retry_delay)
                    await asyncio.sleep(wait)
                else:
                    raise

        raise RuntimeError(f"魹ｽ髃ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾔｴ魹ｽ魹ｽ魹ｽ {self.config.max_retries}")

class MultiTenantQuotaManager:
    """魹ｽ魹ｽ魹ｽ篏ｧ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽﾈｨ魹ｽﾘｷ魹ｽ魹ｽ魹ｽ + 魹ｽ魹ｽ魹ｽﾔｽ魹ｽ魹ｽ魹ｽ"""

    def __init__(self, total_rpm: int):
        self.total_rpm = total_rpm
        self.tenant_limits: dict[str, int] = {}  # 魹ｽ篏ｧ -> 魹ｽ魹ｽ魹ｽ
        self.tenant_usage: dict[str, list[float]] = {}  # 魹ｽ篏ｧ -> 魹ｽ魹ｽ魹ｽ魹ｽﾊｱ魹ｽ魹ｽ魹ｽ

    def set_quota(self, tenant_id: str, weight: float, total_weight: float):
        """魹ｽ魹ｽﾈｨ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ魹ｽ篏ｧ魹ｽ魹ｽ魹ｽ"""

    def allow(self, tenant_id: str) -> bool:

        now = time.time()

        usage = [t for t in usage if t > now - 60]

            usage.append(now)
            self.tenant_usage[tenant_id] = usage
            return True

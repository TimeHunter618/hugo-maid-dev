{{- /* 获取当前激活的风格名，支持 random 随机模式 */ -}} {{- $coversData := hugo.Data.covers -}} {{- $styleName := "" -}} {{- if and $coversData $coversData.styles $coversData.active -}} {{- $active := $coversData.active -}} {{- if eq $active "random" -}} {{- /* 随机模式：从所有风格中随机选一个 */ -}} {{- $styleNames := slice -}} {{- range $name, $_ := $coversData.styles -}} {{- $styleNames = $styleNames | append $name -}} {{- end -}} {{- if gt (len $styleNames) 0 -}} {{- $shuffled := shuffle $styleNames -}} {{- $styleName = index $shuffled 0 -}} {{- end -}} {{- else -}} {{- $styleName = $active -}} {{- end -}} {{- end -}} {{ $styleName -}}
